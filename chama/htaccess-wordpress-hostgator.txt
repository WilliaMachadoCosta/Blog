# Arquivo .htaccess para WordPress no Hostgator
# Este arquivo deve ser colocado na raiz do WordPress no hostgator
# para evitar conflitos com o Vercel

# Permitir acesso apenas às rotas necessárias do WordPress
<IfModule mod_rewrite.c>
RewriteEngine On

# Permitir acesso à API do WordPress
RewriteCond %{REQUEST_URI} ^/wp-json/.*$
RewriteRule ^(.*)$ - [L]

# Permitir acesso aos arquivos de conteúdo do WordPress (mídia, uploads)
RewriteCond %{REQUEST_URI} ^/wp-content/.*$
RewriteRule ^(.*)$ - [L]

# Permitir acesso aos arquivos do sistema do WordPress
RewriteCond %{REQUEST_URI} ^/wp-includes/.*$
RewriteRule ^(.*)$ - [L]

# Permitir acesso ao wp-cron.php
RewriteCond %{REQUEST_URI} ^/wp-cron\.php$
RewriteRule ^(.*)$ - [L]

# Permitir acesso ao xmlrpc.php
RewriteCond %{REQUEST_URI} ^/xmlrpc\.php$
RewriteRule ^(.*)$ - [L]

# Permitir acesso a arquivos .well-known (para SSL, etc.)
RewriteCond %{REQUEST_URI} ^/\.well-known/.*$
RewriteRule ^(.*)$ - [L]

# Para todas as outras rotas, redirecionar para o Vercel
RewriteCond %{REQUEST_URI} !^/wp-json/.*$
RewriteCond %{REQUEST_URI} !^/wp-content/.*$
RewriteCond %{REQUEST_URI} !^/wp-includes/.*$
RewriteCond %{REQUEST_URI} !^/wp-cron\.php$
RewriteCond %{REQUEST_URI} !^/xmlrpc\.php$
RewriteCond %{REQUEST_URI} !^/\.well-known/.*$
RewriteCond %{REQUEST_URI} !^/favicon\.ico$
RewriteCond %{REQUEST_URI} !^/robots\.txt$
RewriteCond %{REQUEST_URI} !^/sitemap\.xml$
RewriteCond %{REQUEST_URI} !^/ads\.txt$
RewriteRule ^(.*)$ https://chamanozap.net/$1 [R=301,L]

# Configurações de segurança
<Files "wp-config.php">
    Order Allow,Deny
    Deny from all
</Files>

<Files ".htaccess">
    Order Allow,Deny
    Deny from all
</Files>

# Headers de segurança
<IfModule mod_headers.c>
    Header always set X-Content-Type-Options nosniff
    Header always set X-Frame-Options SAMEORIGIN
    Header always set X-XSS-Protection "1; mode=block"
    Header always set Referrer-Policy "strict-origin-when-cross-origin"
</IfModule>

# Compressão GZIP
<IfModule mod_deflate.c>
    AddOutputFilterByType DEFLATE text/plain
    AddOutputFilterByType DEFLATE text/html
    AddOutputFilterByType DEFLATE text/xml
    AddOutputFilterByType DEFLATE text/css
    AddOutputFilterByType DEFLATE application/xml
    AddOutputFilterByType DEFLATE application/xhtml+xml
    AddOutputFilterByType DEFLATE application/rss+xml
    AddOutputFilterByType DEFLATE application/javascript
    AddOutputFilterByType DEFLATE application/x-javascript
</IfModule>

# Cache de navegador
<IfModule mod_expires.c>
    ExpiresActive on
    ExpiresByType text/css "access plus 1 year"
    ExpiresByType application/javascript "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
    ExpiresByType image/jpg "access plus 1 year"
    ExpiresByType image/jpeg "access plus 1 year"
    ExpiresByType image/gif "access plus 1 year"
    ExpiresByType image/svg+xml "access plus 1 year"
</IfModule>
